140/** Class representing a list of different code generators.
141 * It offers the same interface as the specific code generators,
142 * and will forward each method to all generators that are enabled.
143 */
144class OutputCodeList
145{
146 public:
147 using OutputCodeVariant = std::variant<HtmlCodeGenerator,
148 HtmlCodeGeneratorDefer,
149 LatexCodeGenerator,
150 LatexCodeGeneratorDefer,
151 RTFCodeGenerator,
152 RTFCodeGeneratorDefer,
153 ManCodeGenerator,
154 ManCodeGeneratorDefer,
155 DocbookCodeGenerator,
156 DocbookCodeGeneratorDefer,
157 XMLCodeGenerator,
158 DevNullCodeGenerator,
159 OutputCodeDeferExtension
160 >;
161
162 int id() const { return m_id; }
163 void setId(int id) { m_id = id; }
164
165 /** Add a code generator already wrapped in a variant type. */
166 void add(OutputCodeVariant &&v)
167 {
168 m_outputCodeList.emplace_back(std::move(v));
169 }
170
171 /** Add a code generator to the list, using a syntax similar to std::make_unique<T>() */
172 template<class T,class... As>
173 T* add(As&&... args)
174 {
175 add(OutputCodeVariant{std::in_place_type<T>,std::forward<As>(args)...});
176 return std::get_if<T>(&m_outputCodeList.back().variant);
177 }
178
179 /** Returns a pointer to a specific generator in the list */
180 template<class T>
181 T *get()
182 {
183 for (auto &e : m_outputCodeList)
184 {
185 T *t = std::get_if<T>(&e.variant);
186 if (t) return t;
187 }
188 return nullptr;
189 }
190
191 /** Enable or disable a specific generator */
192 void setEnabledFiltered(OutputType o,bool enabled)
193 {
194 for (auto &e : m_outputCodeList)
195 {
196 std::visit([&](auto &&arg) { if (arg.type()==o) e.enabled = enabled; },e.variant);
197 }
198 }
199
200 void clear()
201 {
202 m_outputCodeList.clear();
203 }
204
205 // ---- OutputCodeIntf forwarding
206
207 void codify(const QCString &s)
208 { foreach<OutputCodeIntf::codify>(s); }
209
210 void writeCodeLink(CodeSymbolType type,
211 const QCString &ref,const QCString &file,
212 const QCString &anchor,const QCString &name,
213 const QCString &tooltip)
214 { foreach<OutputCodeIntf::writeCodeLink>(type,ref,file,anchor,name,tooltip); }
215
216 void writeLineNumber(const QCString &ref,const QCString &file,const QCString &anchor,
217 int lineNumber, bool writeLineAnchor)
218 { foreach<OutputCodeIntf::writeLineNumber>(ref,file,anchor,lineNumber,writeLineAnchor); }
219
220 void writeTooltip(const QCString &id, const DocLinkInfo &docInfo, const QCString &decl,
221 const QCString &desc, const SourceLinkInfo &defInfo, const SourceLinkInfo &declInfo)
222 { foreach<OutputCodeIntf::writeTooltip>(id,docInfo,decl,desc,defInfo,declInfo); }
223
224 void startCodeLine(bool hasLineNumbers)
225 { foreach<OutputCodeIntf::startCodeLine>(hasLineNumbers); }
226
227 void endCodeLine()
228 { foreach<OutputCodeIntf::endCodeLine>(); }
229
230 void startFontClass(const QCString &c)
231 { foreach<OutputCodeIntf::startFontClass>(c); }
232
233 void endFontClass()
234 { foreach<OutputCodeIntf::endFontClass>(); }
235
236 void writeCodeAnchor(const QCString &name)
237 { foreach<OutputCodeIntf::writeCodeAnchor>(name); }
238
239 void startCodeFragment(const QCString &style)
240 { foreach<OutputCodeIntf::startCodeFragment>(style); }
241
242 void endCodeFragment(const QCString &style)
243 { foreach<OutputCodeIntf::endCodeFragment>(style); }
244
245 private:
246 template<template <class> class GeneratorT, class... As>
247 void foreach(As&&... args)
248 {
249 for (auto &e : m_outputCodeList)
250 {
251 if (e.enabled)
252 {
253 dispatch_call<GeneratorT>(e.variant,std::forward<As>(args)...);
254 }
255 }
256 }

260 explicit OutputCodeElem(OutputCodeVariant &&v) : variant(std::move(v)) {}
261 OutputCodeVariant variant;
262 bool enabled = true;
263 };
264
265 std::vector<OutputCodeElem> m_outputCodeList;
266 int m_id = -1;
