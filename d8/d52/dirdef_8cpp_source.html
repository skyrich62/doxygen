112DirDefImpl::DirDefImpl(const QCString &path) : DefinitionMixin(path,1,1,path)
113{
114 bool fullPathNames = Config_getBool(FULL_PATH_NAMES);
115 // get display name (stripping the paths mentioned in STRIP_FROM_PATH)
116 // get short name (last part of path)
117 m_shortName = path;
118 m_diskName = path;
119 if (m_shortName.at(m_shortName.length()-1)=='/')
120 { // strip trailing /
121 m_shortName = m_shortName.left(m_shortName.length()-1);
122 }
123 int pi=m_shortName.findRev('/');
124 if (pi!=-1)
125 { // remove everything till the last /
126 m_shortName = m_shortName.mid(pi+1);
127 }
128 setLocalName(m_shortName);
129 m_dispName = fullPathNames ? stripFromPath(path) : m_shortName;
130 if (m_dispName.length()>0 && m_dispName.at(m_dispName.length()-1)=='/')
131 { // strip trailing /
132 m_dispName = m_dispName.left(m_dispName.length()-1);
133 }
134
135 m_level=-1;
136 m_parent=0;
137}

655/** Add as "uses" dependency between \a this dir and \a dir,
656 * that was caused by a dependency on file \a fd.
657 * srcDirect and dstDirect indicate if it is a direct dependencies (true) or if
658 * the dependencies was indirect (e.g. a parent dir that has a child dir that has the dependencies)
659 */
660void DirDefImpl::addUsesDependency(const DirDef *dir,const FileDef *srcFd,
661 const FileDef *dstFd,bool srcDirect, bool dstDirect)
662{
663 if (this==dir) return; // do not add self-dependencies
664 AUTO_TRACE("add dependency {}->{} due to {}->{}",
665 qPrint(shortName()),
666 qPrint(dir->shortName()),
667 qPrint(srcFd->name()),
668 qPrint(dstFd->name()));
669
670 // levels match => add direct dependency
671 bool added=FALSE;
672 UsedDir *usedDir = m_usedDirs.find(dir->getOutputFileBase());
673 if (usedDir) // dir dependency already present
674 {
675 const FilePair *usedPair = usedDir->findFilePair(FilePair::key(srcFd,dstFd));
676 if (usedPair==0) // new file dependency
677 {
678 AUTO_TRACE_ADD("{} => {} new file dependency",srcFd->name(),dstFd->name());
679 usedDir->addFileDep(srcFd,dstFd, srcDirect, dstDirect);
680 added=TRUE;
681 }
682 else
683 {
684 // dir & file dependency already added
685 }
686 }
687 else // new directory dependency
688 {
689 AUTO_TRACE_ADD("{} => {} new file dependency",srcFd->name(),dstFd->name());
690 auto newUsedDir = std::make_unique<UsedDir>(dir);
691 newUsedDir->addFileDep(srcFd,dstFd, srcDirect, dstDirect);
692 m_usedDirs.add(dir->getOutputFileBase(),std::move(newUsedDir));
693 added=TRUE;
694 }
695 if (added)
696 {
697 if (dir->parent())
698 {
699 // add relation to parent of used dir
700 addUsesDependency(dir->parent(),
701 srcFd,
702 dstFd,
703 srcDirect,
704 false); // indirect dependency on dest dir
705 }
706 if (parent())
707 {
708 // add relation for the parent of this dir as well
709 parent()->addUsesDependency(dir,
710 srcFd,
711 dstFd,
712 false, // indirect dependency from source dir
713 dstDirect);
714 }
715 }
716}

718/** Computes the dependencies between directories
719 */
720void DirDefImpl::computeDependencies()
721{
722 AUTO_TRACE();
723 for (const auto &fd : m_fileList)
724 {
725 AUTO_TRACE_ADD("dir={} file={}",shortName(),fd->name());
726 for (const auto &ii : fd->includeFileList())
727 {
728 AUTO_TRACE_ADD("#include {}",ii.includeName);
729 if (ii.fileDef && ii.fileDef->isLinkable()) // linkable file
730 {
731 DirDef *usedDir = ii.fileDef->getDirDef();
732 if (usedDir)
733 {
734 // add dependency: thisDir->usedDir
735 AUTO_TRACE_ADD("add dependency {}->{}",name(),usedDir->name());
736 addUsesDependency(usedDir,fd,ii.fileDef,true,true);
737 }
738 }
739 }
740 }
741
742 std::sort(m_usedDirs.begin(),m_usedDirs.end(),
743 [](const auto &u1,const auto &u2)
744 { return qstricmp(u1->dir()->getOutputFileBase(),u2->dir()->getOutputFileBase())<0; });
745
746 for (const auto& usedDirectory : m_usedDirs)
747 {
748 usedDirectory->sort();
749 }

779void UsedDir::addFileDep(const FileDef *srcFd,const FileDef *dstFd, bool srcDirect, bool dstDirect)
780{

947/** In order to create stable, but unique directory names,
948 * we compute the common part of the path shared by all directories.
949 */
950static void computeCommonDirPrefix()
951{
952 AUTO_TRACE();
953 QCString path;
954 auto it = Doxygen::dirLinkedMap->begin();
955 if (!Doxygen::dirLinkedMap->empty()) // we have at least one dir
956 {
957 // start will full path of first dir
958 path=(*it)->name();
959 int i=path.findRev('/',path.length()-2);
960 path=path.left(i+1);
961 bool done=FALSE;
962 if (i==-1)
963 {
964 path="";
965 }
966 else
967 {
968 while (!done)
969 {
970 uint32_t l = path.length();
971 size_t count=0;
972 for (const auto &dir : *Doxygen::dirLinkedMap)
973 {
974 QCString dirName = dir->name();
975 if (dirName.length()>path.length())
976 {
977 if (dirName.left(l)!=path) // dirName does not start with path
978 {
979 i=path.findRev('/',l-2);
980 if (i==-1) // no unique prefix -> stop
981 {
982 path="";
983 done=TRUE;
984 }
985 else // restart with shorter path
986 {
987 path=path.left(i+1);
988 break;
989 }
990 }
991 }
992 else // dir is shorter than path -> take path of dir as new start
993 {
994 path=dir->name();
995 l=path.length();
996 i=path.findRev('/',l-2);
997 if (i==-1) // no unique prefix -> stop
998 {
999 path="";
1000 done=TRUE;
1001 }
1002 else // restart with shorter path
1003 {
1004 path=path.left(i+1);
1005 }
1006 break;
1007 }
1008 count++;
1009 }
1010 if (count==Doxygen::dirLinkedMap->size())
1011 // path matches for all directories -> found the common prefix
1012 {
1013 done=TRUE;
1014 }
1015 }
1016 }
1017 }
1018 for (const auto &dir : *Doxygen::dirLinkedMap)
1019 {
1020 QCString diskName = dir->name().right(dir->name().length()-path.length());
1021 dir->setDiskName(diskName);
1022 AUTO_TRACE_ADD("set disk name: {} -> {}",dir->name(),diskName);
1023 }
