135bool OutputList::isEnabled(OutputType o)
136{
137 bool found = false;
138 bool enabled = false;
139 for (const auto &e : m_outputGenList)
140 {
141 std::visit([&](auto &&gen) {
142 if (gen.type()==o) { enabled = e.enabled; found=true; }
143 }, e.variant);
144 if (found) return enabled;
145 }
146 return enabled;
147}

159void OutputList::popGeneratorState()
160{
161 //printf("popGeneratorState()\n");
162 for (auto &e : m_outputGenList)
163 {
164 if (!e.enabledStack.empty())
165 {
166 e.enabled = e.enabledStack.top();
167 e.enabledStack.pop();
168 }
169 }
170 syncEnabled();

173void OutputList::generateDoc(const QCString &fileName,int startLine,
174 const Definition *ctx,const MemberDef * md,
175 const QCString &docStr,bool indexWords,
176 bool isExample,const QCString &exampleName,
177 bool singleLine,bool linkFromIndex,
178 bool markdownSupport)
179{
180 if (docStr.isEmpty()) return;
181
182 auto count=std::count_if(m_outputGenList.begin(),m_outputGenList.end(),
183 [](const auto &e) { return e.enabled; });
184 if (count>0)
185 {
186 // we want to validate irrespective of the number of output formats
187 // specified as:
188 // - when only XML format there should be warnings as well (XML has its own write routines)
189 // - no formats there should be warnings as well
190 auto parser { createDocParser() };
191 auto ast { validatingParseDoc(*parser.get(),
192 fileName,startLine,
193 ctx,md,docStr,indexWords,isExample,exampleName,
194 singleLine,linkFromIndex,markdownSupport) };
195 if (ast) writeDoc(ast.get(),ctx,md);
196 }
197}
